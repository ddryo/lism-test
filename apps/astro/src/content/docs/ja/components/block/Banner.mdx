import { Banner, Center, Layer, MediaLayer, FilterLayer } from '@lism/astro';
import LipsumText from '~/components/LipsumText.astro';
import { Image } from 'astro:assets';

↓ media に画像データを指定
<Banner 
	href="#banner-link"
	ratio="16/9"
	c="white"
	p={40}
	media={{ tag: 'img', src: "/img/demo-img.jpeg", alt: "", width: 960, height: 640 }}
	filter={{ bgc: "rgba(0, 0, 0, 0.5)" }}
>
	<Center size="cover">
		<LipsumText long />
	</Center>
</Banner>


↓ media に Image コンポーネントを指定
<Banner
	href="#banner-link"
	ratio="2/1"
	c="white"
	p={40}
	media={{ tag: Image, src: "/img/demo-img.jpeg", alt: "", width: 960, height: 640 }}
	filter={{ bgc: "rgba(0, 0, 0, 0.5)" }}
>
	<Center size="cover">
		<LipsumText long />
	</Center>
</Banner>


`<MediaLayer>`, `<FilterLayer>` 直接使うことも可能

<Banner
	href="#banner-link"
	ratio="16/9"
	c="white"
	p={40}
>
	<MediaLayer>
		<Image src="/img/demo-img.jpeg" alt="" width={960} height={640} />
	</MediaLayer>
	<FilterLayer bgc="rgba(0, 0, 0, 0.5)" blur="4px"/>
	<Center size="cover">
		<LipsumText long />
	</Center>
</Banner>

<Banner
