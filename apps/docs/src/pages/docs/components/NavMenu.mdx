# NavMenu


## Props

- `provide.p`でpaddingを一括コントロールできます。


## Import

```js
import { NavMenu } from '@loos/lism-core';
```


## DEMO

export const Items = ({ length = 3, ...props }) => {
	// Array.from({length}).map((i) => {
	// Array.from(Array(ct))
	// Array.from(Array(ct))
	return [...Array(length)].map((_,i) => (
		<NavMenuItem key={i}>
			<NavMenuLink href={`#menu${i+1}`} {...props}>Lorem item</NavMenuLink>
		</NavMenuItem> 
	));
};

export const DemoBadge = ({ num = "10", ml = "auto" }) => {
	return (
		<Badge fz="12px" ml={ml} fw="400" color="dimgray">
			{num}
		</Badge>
	);
};
