export const NoteDemos = ({ children, variant }) => {
	const PRESETS = ["alert", "point", "warning", "check", "info", "help", "note"];
	return (
		<>
			{PRESETS.map((preset) => {
				return (
					<Note heading="Note default" variant={variant} preset={`${preset}`} key={preset}>
						<DemoText lang="ja"/>
					</Note>
				);
			})}
		</>
	);
};
