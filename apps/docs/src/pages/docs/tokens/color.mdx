import { Box, Text, RatioGrid, Flex, Stack, Spacer } from '@loos/lism-core';

export const LismColorDemos = ({ colors = [] }) => {
	return (
		<>
			{colors.map((name) => {
				return (
					<Stack key={name} gap="10">
						<Text bd='left' bdw='2px' lh='1.25' pl="20" fz='xs'><code>--{name}</code></Text>
						<RatioGrid ratio={['1', '1:2']} fz='s'>
							<Box bgc={`var(--${name})`} p="10">&emsp;</Box>
							<Box bd='' c={`var(--${name})`} p="10">Lorem ipsum text...</Box>
						</RatioGrid>
					</Stack>
				);
			})}
		</>
	);
};

## セマンティックトークン

### 文字色(`color`)用 
|トークン|変数|初期値|
|---|---|---|
|`base`|`--c--base`|`--b900`|
|`pale`|`--c--pale`|`--b700`|
|`opposite`|`--c--opposite`|`--b50`|

weak - srtong?
pale - deep?

### 背景色(`background-color`)用 
|トークン|変数|初期値|
|---|---|---|
|`base`|`--bgc--base`|`--b50`|
|`pale`|`--bgc--pale`|`--b100`|
|`opposite`|`--bgc--opposite`|`--b900`|

### ボーダー色(`border-color`)用 
|トークン|変数|初期値|
|---|---|---|
|`base`|`--bdc--base`|`--b300`|
{/* |`strong`?|`--bdc--strong?`|`--b800`| */}
